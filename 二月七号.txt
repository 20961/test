#include<stdio.h>
int main()
{
    int ret = 0;
    int ch = 0;
    char password[20] = { 0 };
    printf("请输入密码：>");
    scanf("%s", password);//输入密码，并存取password数组中
    //缓冲区中还剩余一个‘\n’
    //读取一下'\n'
    while ((ch = getchar()) != '\n')
    {
        ;
    }
    printf("请确认（Y/N):>");
    ret = getchar();//Y/N
    if (ret == 'Y')
    {
        printf("确认成功\n");
    }
    else
    {
        printf("放弃确认\n");
    }
    return 0;
}
